204  void fillGpuInfo(int deviceNumber) {
205 #if defined(VoxxCudaSupported)
206  int deviceCount = 0;
207  VoxxCudaErrorCheck(cudaGetDeviceCount(&deviceCount));
208  if (deviceNumber >= deviceCount) {
209  printf(
210  "\nError: Device: %03i is larger than the %03i Gpus in the system\n",
211  deviceNumber, deviceCount);
212  return;
213  }
214 
215  cudaSetDevice(deviceNumber);
216  cudaDeviceProp deviceProps;
217  VoxxCudaErrorCheck(cudaGetDeviceProperties(&deviceProps, deviceNumber));
218 
219  // Rewritten version from cuda samples:
220  auto coresForArch = [] (int major, int minor) {
221  enum { Fermi = 2, Kepler = 3, Maxwell = 5, Pascal = 6 };
222  switch (major) {
223  case Fermi : return minor == 0 ? 32 : 48;
224  case Kepler : return 192;
225  case Maxwell : return 128;
226  case Pascal : return minor == 0 ? 64 : 128;
227  default : return -1;
228  }
229  };
230  setArchitecture((deviceProps.major << 4) + deviceProps.minor);
231  setMultiprocessors(deviceProps.multiProcessorCount);
232  setCoresPerMultiprocessor(coresForArch(deviceProps.major,
233  deviceProps.minor));
234 #endif // VoxxCudaSupported
235  }
