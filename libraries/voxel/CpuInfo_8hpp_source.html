124 static inline uint32_t apicid() noexcept {
125  auto regs = cpuid(Function::MaxLeaf);
126  // If leaf b is supported, then we can return x2APPIC ID:
127  if (regs.eax() >= Function::LeafB)
128  return cpuid(Function::LeafB).edx();

131  return getBits(cpuid(Function::Features).ebx(), 24, 31);
