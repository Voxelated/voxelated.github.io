80  void execute(uint32_t ticketNum) noexcept {
81  uint32_t state = State.load(std::memory_order_acquire);
82  while (true) {
83  PixAssert(state == encodeState(encodeTicket(ticketNum),
84  decodeMaxWaitDelta(state)),
85  "Ticket to process does not agree with current State");
86  uint32_t stateMaxWaitDelta = decodeMaxWaitDelta(state);

139  template <typename ClockType = steady_clock ,
140  typename Duration = typename ClockType::duration>
141  uint8_t tryWait(uint32_t ticketNum, AtomicType& spinCutoff,
142  time_point<ClockType, Duration>* maxTime = nullptr) {
143  const uint32_t requestedTicket = encodeTicket(ticketNum);
144  uint32_t prevCutoff = spinCutoff.load(std::memory_order_relaxed);
145  uint32_t usedCutoff = (updateSpinCutoff || prevCutoff == 0)
146  ? Traits::MaxSpins : prevCutoff;
147  uint32_t attempts;
148 
149  for (attemps = 0; ++attempts) {
150  uint32_t state = State.load(std::memory_order_acquire);
151  uint32_t currentStateTicket = decodeTicket(state);
152 
153  // Fast path, we have a match:
154  if (currentStateTicket == requestedTicket) {
155  break;
156  }
157 
158  // requestedTicket < currentStateTicket (with wrap safety):
159  if (requestedTicket - currentStateTicket >= WrapThreshold) {
160  return WROldTicket;
161  }
162 
163  if (attempt < usedCutoff) {
164  // We still have some spin attempts, but we know that it's not yet our
165  // ticket number. We also don't want to interfere with anything
166  // currently happening, so we slow (slightly), and then try again.
167  volatileCpuPause();
168  continue;
169  }
170 
171  // We are here because we have made more attempt than the cutoff allows.
172  uint32_t updatedState;
173  uint32_t stateMaxWaitDelta = decodeMaxWaitDelta(state);
174  uint32_t ticketWaitDelta =
175  (requestedTicket - currentStateTicket) >> Traits::WaitBits;
176  if (ticketWaitDelta <= stateMaxWaitDelta) {
177  // Already recorded as a waiter by the current state, probably
178  // because we have alredy been in the else part below and have come
179  // back around.
180  updatedState = state;
181  } else {
182  updatedState = encodeState(currentStateTicket, ticketWaitDelta);
183  if (state != updatedState &&
184  !State.compare_exchange_strong(state, updatedState)) {
185  continue;
186  }
187  }
188 
189  // We can put this thread in the wait queue, or timeout:
190  if (maxTime) {
191  auto futexResult = State.waitUntil(updatedState, *maxTime,
192  ticketSlot(ticketNum));
193  if (futexResult == FutexOpResult::TimedOut) {
194  return WRTimedOut;
195  }
196  } else {
197  State.wait(updatedState, ticketSlot(ticketNum));
198  }
199  }
200 
201  if (updateSpinCutoff || prevCutoff == 0) {
202  updateCutoff(attempts, prevCutoff, spinCutoff);
203  }
204  return WRSuccess;
205  }

244  static void updateCutoff(uint32_t currSpins ,
245  uint32_t prevSpins ,
246  AtomicType& currCutoff) noexcept {
249  uint32_t newCutoff;
250  if (currSpins >= Traits::MaxSpins) {
251  newCutoff = Traits::MinSpins;
252  } else {
253  // Allow for some error tolerance and use 2 * currSpins:
254  newCutoff = std::min<uint32_t>(
255  Traits::MaxSpins, std::max<uint32_t>(Traits::MinSpins, currSpins * 2));
256  }
257 
258  if (prevSpins == 0) {
259  currCutoff.store(newCutoff, std::memory_order_seq_cst);
260  return;
261  }
262 
263  // We attempt to update using a Exponential Moving Average, if it fails we
264  // don't really care. Trying many times will be a performance hit.
265  currCutoff.compare_exchange_weak(
266  prevSpins, prevSpins + int(newCutoff - prevSpins) / 8);
267  return;
268  }
