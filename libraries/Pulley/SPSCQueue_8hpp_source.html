183  bool push(ElementArgs&&... elemArgs) {
184  const auto currentWriteIdx = WriteIndex.load(std::memory_order_relaxed);
185  auto nextElementIdx = currentWriteIdx + 1;
186  if (nextElementIdx == Capacity) {

306 template <typename T>
307 SPSCQueue<T>::SPSCQueue(SPSCQueue<T>&& other)
308 : Capacity(other.Capacity) ,
309  Elements(static_cast<ValueType* const>(
310  std::malloc(sizeof(ValueType) * other.Capacity))) {
311  // Here we ensure that the loads are sequentially consistent so that the
312  // next operations using the variables use the correct values.
313  ReadIndex.store(other.ReadIndex.load(std::memory_order_relaxed));
314  WriteIndex.store(other.WriteIndex.load(std::memory_order_relaxed));
315 
316  auto currentIdx = ReadIndex.load(std::memory_order_relaxed);
317  const auto endIdx = WriteIndex.load(std::memory_order_relaxed);
318 
319  while (currentIdx != endIdx) {
320  if (currentIdx == Capacity) {
321  if (endIdx == 0) {
322  break; // currentIdx = end element, endIdx = 0 element
323  }
324  currentIdx = 0;
325  }
326  Elements[currentIdx] = std::move(other.Elements[currentIdx]);
327  other.Elements[currentIdx].~ValueType();
328  ++currentIdx;
329  }
330 }
