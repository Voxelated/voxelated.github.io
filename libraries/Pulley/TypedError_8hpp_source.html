558 template <typename... HandlerTypes>
559 TypedError catchTypedErrors(TypedError error, HandlerTypes... handlers) {
560  if (!error) {
561  return TypedError();
562  }
563 
564  error.Checked = true;
565  if (!error.ErrorInfo->isClass(TypedErrorList::classID())) {
566  return Detail::handleTypedError(std::move(error), handlers...);
567  }
568 
569  // List of errors, need to try and handle each case:
570  using UniqueList = std::unique_ptr<TypedErrorList>;
571  using InfosContainer = typename TypedErrorList::ErrorInfoContainer;
572 
573  UniqueList errorList(static_cast<TypedErrorList*>(error.ErrorInfo.release()));
574  InfosContainer errorInfos = std::move(errorList->ErrorInfos);
575  errorList->ErrorInfos = InfosContainer();
576 
577  for (auto& errInfo : errorInfos) {
578  if (auto err = Detail::handleTypedError(std::move(errInfo), handlers...)) {
579  err.Checked = true;
580  errorList->ErrorInfos.push_back(std::move(err.ErrorInfo));
581  }
582  }
583 
584  if (errorList->ErrorInfos.empty()) {
585  return TypedError();
586  }
587 
588  return TypedError(std::move(errorList));
589 }
