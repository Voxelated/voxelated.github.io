110  template <typename... ObjectArgs>
111  void push(ObjectArgs&&... args) {
112  const auto bottom = Bottom.load(std::memory_order_relaxed);
113  const auto index = wrappedIndex(bottom);
114  new (&Objects[index]) Object(std::forward<ObjectArgs>(args)...);
115 
116  // Ensure that the compiler does not reorder this instruction and the
117  // setting of the object above, otherwise the object seems added (since the
118  // bottom index has moved) but isn't (since it would not have been added).
119  Bottom.store(bottom + 1, std::memory_order_release);
120  }

152  OptionalType pop() {
153  using namespace std::experimental;
154  auto bottom = Bottom.load(std::memory_order_relaxed) - 1;
155 
156  // Sequentially consistant memory ordering is used here to ensure that the
157  // load to top always happens after the load to bottom above, and that the
158  // compiler emits an __mfence__ instruction. Unfortunately, benchmarking has
159  // shown that the mfence does degrade performance, but the code obviously
160  // has to be correct.
161  Bottom.store(bottom, std::memory_order_seq_cst);
162 
163  auto top = Top.load(std::memory_order_relaxed);
164  if (top > bottom) {
165  Bottom.store(top, std::memory_order_relaxed);
166  return nullopt;
167  }
168 
169  auto object = make_optional(Objects[wrappedIndex(bottom)]);
170  if (top != bottom)
171  return object;
172 
173  // If we are here there is only one element left in the queue, and there may
174  // be a race between this method and steal() to get it. If this exchange is
175  // true then this method won the race (or was not in one) and then the
176  // object can be returned, otherwise the queue has already been emptied.
177  bool exchanged = Top.compare_exchange_strong(top ,
178  top + 1 ,
179  std::memory_order_release);
180 
181  // This is also a little tricky: If we lost the race, top will be changed to
182  // the new value set by the stealing thread (i.e it's already incremented).
183  // If it's incremented again then Bottom > Top when the last item was
184  // actually just cleared. This is also the unlikely case -- since this path
185  // is only executed when there is contention on the last element -- so the
186  // const of the branch is acceptable.
187  Bottom.store(top + (exchanged ? 1 : 0), std::memory_order_relaxed);
188 
189  return exchanged ? object : nullopt;
190  }

214  OptionalType steal() {
215  using namespace std::experimental;
216  // Top must be loaded before bottom, so we use acquire ordering since it
217  // ensures that everyting below it stays below it.
218  auto top = Top.load(std::memory_order_relaxed);
219 
220  // Top must always be set before bottom, so that bottom - top represents an
221  // accurate enough (to prevent error) view of the queue size. Loads to
222  // different address aren't reordered (i.e load load barrier)
223  Thread::memoryBarrier();
224  auto bottom = Bottom.load(std::memory_order_relaxed);
225 
226  if (top >= bottom) {
227  return nullopt;
228  }
229 
230  // Here the object at top is fetched, and and update to Top is atempted,
231  // since the top element is being stolen. __If__ the exchange succeeds,
232  // then this method won a race with another thread to steal the element,
233  // or if there is only a single element left, then it won a race between
234  // other threads and the pop() method (potentially), or there was no race.
235  // In summary, if the exchange succeeds, the object can be returned,
236  // otherwise it can't.
237  //
238  // Also note that the object __must__ be created before the exchange to top,
239  // otherwise there will potentially be a race to construct the object and
240  // between a thread pushing onto the queue.
241  //
242  // This introduces overhead when there is contention to steal, and the steal
243  // is unsuccessful, but in the succesful case there is no overhead.
244  auto object = make_optional(Objects[wrappedIndex(top)]);
245  bool exchanged = Top.compare_exchange_strong(top ,
246  top + 1 ,
247  std::memory_order_release);
248 
249  return exchanged ? object : nullopt;
250  }
