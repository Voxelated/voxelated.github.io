146  OptionalType pop() {
147  using namespace std::experimental;
148  auto bottom = Bottom.load(std::memory_order_relaxed) - 1;
149  Bottom.store(bottom, std::memory_order_relaxed);
150 
151  // This acts as a barrier to ensure that top is always set __after__ bottom,
152  // and that neither the compiler nor the cpu reorder the instructions.
153 
154  // All non-relaxed atomic operations cause ordering within their own thread,
155  // so this acquire operation is a barrier between it and the store above.
156  auto top = Top.load(std::memory_order_acquire);
157 
158  if (top > bottom) {
159  Bottom.store(top, std::memory_order_relaxed);
160  return OptionalType{};
161  }
162 
163  auto object = make_optional(Objects[wrappedIndex(bottom)]);
164  if (top != bottom)
165  return object;
166 
167  // If we are here there is only one element left in the queue, and there may
168  // be a race between this method and steal() to get it. If this exchange is
169  // true then this method won the race (or was not in one) and then the
170  // object can be returned, otherwise the queue has already been emptied.
171  bool exchanged = Top.compare_exchange_strong(top ,
172  top + 1 ,
173  std::memory_order_release);
174 
175  Bottom.store(top + 1, std::memory_order_relaxed);
176  return exchanged ? object : OptionalType{};
177  }
178 
208  OptionalType steal() {
209  using namespace std::experimental;
210  auto top = Top.load(std::memory_order_acquire);
211 
212  // The top variable must be loaded __before__ bottom, incase pop()'s from
213  // the queue's thread have modified the queue.
214  auto bottom = Bottom.load(std::memory_order_relaxed);
215 
216  if (top >= bottom)
217  return OptionalType{};
218 
219  // Here the object at top is fetched, and and update to Top is atempted,
220  // since the top element is being stolen. __If__ the exchange succeeds,
221  // then this method won a race with another thread to steal the element,
222  // or if there is only a single element left, then it won a race between
223  // other threads and the pop() method (potentially), or there was no race.
224  // In summary, if the exchange succeeds, the object can be returned,
225  // otherwise it can't.
226  auto object = make_optional(Objects[wrappedIndex(top)]);
227  bool exchanged = Top.compare_exchange_strong(top ,
228  top + 1 ,
229  std::memory_order_release);
230 
231  return exchanged ? object : OptionalType{};
232  }
